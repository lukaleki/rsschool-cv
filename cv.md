## Code Example

```javascript
function moveZeros(arr) {
  let myArr = [];
  let zeroCount = 0;
  let arrLength = arr.length;

  for (let i = 0; i < arrLength; i++) {
    if (arr[i] === 0) {
      zeroCount++;
    } else {
      myArr.push(arr[i]);
    }
  }

  for (let i = 0; i < zeroCount; i++) {
    myArr.push(0);
  }

  return myArr;
}
```
